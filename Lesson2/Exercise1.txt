let a = 1, b = 1, c, d; // Здесь  объявлены четыре переменные, двум из которых присвоены значения.

c = ++a; // Здесь переменной c присвоено значение 2, так как сначала выполняется операция ++ а затем происходит присовение значения. Т.о. c == 2, a == 2

d = b++; // В данном случае переменной d присваивается значение 1, так как при такой записи сначала выполняется присвоение, а затем переменная b увеличивается на 1. 
         // Т.о. b == 2, d == 1

c = 2 + ++a; // Здесь по аналогии a увеличивается на 1 и складывается с 2. Таким образом c == 5, a == 3

d = 2 + b++; // Два складываем со значением b, это значение присваивается d, затем b увеличивается на 1. Т.о. d == 4, b == 3

alert(a); // 3

alert(b); // 3
